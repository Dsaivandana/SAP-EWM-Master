METHOD GET_MATERIAL_DETAILS.

  IF IM_MATNR IS INITIAL .
    RAISE EVENT NO_MATERIAL.
  ELSE.
    SELECT SINGLE * FROM MARA
      INTO EX_MARA
      WHERE MATNR = IM_MATNR.
  ENDIF.

ENDMETHOD.
