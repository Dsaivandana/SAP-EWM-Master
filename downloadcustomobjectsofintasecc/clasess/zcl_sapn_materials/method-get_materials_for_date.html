method GET_MATERIALS_FOR_DATE.

*  *Get material no, created date, material type, material group, units of measue
*from MARA table
  SELECT MATNR ERSDA MTART MATKL MEINS
    FROM MARA INTO TABLE ET_MARA
    WHERE ERSDA = IM_DATE.

endmethod.
