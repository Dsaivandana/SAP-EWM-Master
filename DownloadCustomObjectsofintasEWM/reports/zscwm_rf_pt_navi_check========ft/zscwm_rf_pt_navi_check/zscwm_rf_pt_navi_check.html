FUNCTION zscwm_rf_pt_navi_check.
*"--------------------------------------------------------------------
*"*"Local Interface:
*"  CHANGING
*"     REFERENCE(CS_PTWY) TYPE  /SCWM/S_RF_PTWY
*"     REFERENCE(CS_ADMIN) TYPE  /SCWM/S_RF_ADMIN
*"     REFERENCE(CT_PTWY) TYPE  /SCWM/TT_RF_PTWY
*"     REFERENCE(CT_LGPLA) TYPE  /SCWM/TT_LGPLA
*"--------------------------------------------------------------------
  DATA: lv_fcode    TYPE /scwm/de_fcode.

  IF cs_ptwy-nlpla EQ cs_ptwy-nlpla_orig
     OR cs_ptwy-nlpla_orig IS INITIAL.
*    If Dest. Bin is provided then normal process
    lv_fcode = 'PBO1'.
  ELSE.
*    if bin is NOT provided then skip the verification
    lv_fcode = 'ENTER'.
  ENDIF.

* set fcode
  CALL METHOD /scwm/cl_rf_bll_srvc=>set_fcode
    EXPORTING
      iv_fcode = lv_fcode.

* set processing mode
  CALL METHOD /scwm/cl_rf_bll_srvc=>set_prmod
    EXPORTING
      iv_prmod = /scwm/cl_rf_bll_srvc=>c_prmod_background
*     iv_ovr_cust = space
    .

ENDFUNCTION.
